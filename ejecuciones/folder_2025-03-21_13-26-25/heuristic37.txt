
def heuristic(input_data):
    """Combines SPT, machine load balancing, and job completion time."""
    n_jobs = input_data['n_jobs']
    n_machines = input_data['n_machines']
    jobs = input_data['jobs']

    schedule = {}
    machine_time = {m: 0 for m in range(n_machines)}
    machine_load = {m: 0 for m in range(n_machines)}
    job_completion_time = {j: 0 for j in range(1, n_jobs + 1)}

    ready_operations = []
    for job_id, operations in jobs.items():
        ready_operations.append((job_id, 0))

    while ready_operations:
        best_job, best_op_idx, best_machine, best_start_time, best_processing_time = None, None, None, float(
            'inf'), None

        for job_id, op_idx in ready_operations:
            machines, times = jobs[job_id][op_idx]

            for m_idx, machine in enumerate(machines):
                processing_time = times[m_idx]
                start_time = max(machine_time[machine], job_completion_time[job_id])
                end_time = start_time + processing_time
                balancing_factor = machine_load[machine] * 0.001  # Reduced balancing factor
                adjusted_end_time = end_time + balancing_factor

                if adjusted_end_time < best_start_time:  # Using adjusted end time for machine choice
                    best_job, best_op_idx, best_machine, best_start_time, best_processing_time = job_id, op_idx, machine, start_time, processing_time

        job_id = best_job
        op_idx = best_op_idx
        machine = best_machine
        start_time = best_start_time
        processing_time = best_processing_time

        end_time = start_time + processing_time

        if job_id not in schedule:
            schedule[job_id] = []

        schedule[job_id].append({
            'Operation': op_idx + 1,
            'Assigned Machine': machine,
            'Start Time': start_time,
            'End Time': end_time,
            'Processing Time': processing_time
        })

        machine_time[machine] = end_time
        machine_load[machine] += processing_time
        job_completion_time[job_id] = end_time

        ready_operations.remove((job_id, op_idx))

        if op_idx + 1 < len(jobs[job_id]):
            ready_operations.append((job_id, op_idx + 1))

    return schedule
