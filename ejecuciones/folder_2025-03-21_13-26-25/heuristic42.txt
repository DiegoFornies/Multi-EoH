
def heuristic(input_data):
    """Combines priority scheduling, SPT, and load balancing for FJSSP."""

    n_jobs = input_data['n_jobs']
    n_machines = input_data['n_machines']
    jobs = input_data['jobs']

    schedule = {}
    machine_available_times = {m: 0 for m in range(n_machines)}
    job_last_end_times = {j: 0 for j in range(1, n_jobs + 1)}
    job_priorities = {j: 0 for j in range(1, n_jobs + 1)}
    operation_counter = {job: 0 for job in range(1, n_jobs + 1)}

    ready_operations = []
    for job_id, operations in jobs.items():
        ready_operations.append((job_id, 0))

    while ready_operations:
        best_job, best_op_idx, best_machine, best_start_time, best_processing_time = None, None, None, float('inf'), None

        for job_id, op_idx in ready_operations:
            machines, times = jobs[job_id][op_idx]
            
            # SPT based machine selection: consider shortest processing time first
            best_local_machine = None
            best_local_start_time = float('inf')
            best_local_processing_time = None
            for m_idx, machine in enumerate(machines):
                processing_time = times[m_idx]
                start_time = max(machine_available_times[machine], job_last_end_times[job_id])

                if start_time < best_local_start_time:
                    best_local_start_time = start_time
                    best_local_machine = machine
                    best_local_processing_time = processing_time

            if best_local_machine is None:
                continue
                
            # Priority based job selection
            priority = best_local_start_time + job_priorities[job_id]

            if best_start_time is None or priority < best_start_time + job_priorities.get(best_job,0):
                best_job = job_id
                best_op_idx = op_idx
                best_machine = best_local_machine
                best_start_time = best_local_start_time
                best_processing_time = best_local_processing_time

        if best_job is None:
            break

        job_id = best_job
        op_idx = best_op_idx
        machine = best_machine
        start_time = best_start_time
        processing_time = best_processing_time
        end_time = start_time + processing_time

        if job_id not in schedule:
            schedule[job_id] = []

        operation_number = operation_counter[job_id] + 1
        operation_counter[job_id] = operation_number

        schedule[job_id].append({
            'Operation': operation_number,
            'Assigned Machine': machine,
            'Start Time': start_time,
            'End Time': end_time,
            'Processing Time': processing_time
        })

        machine_available_times[machine] = end_time
        job_last_end_times[job_id] = end_time

        ready_operations.remove((job_id, op_idx))

        if op_idx + 1 < len(jobs[job_id]):
            ready_operations.append((job_id, op_idx + 1))

        # Job aging mechanism
        for job_id_priority in job_priorities:
            if job_id_priority != best_job and job_id_priority in [op[0] for op in ready_operations]:
                job_priorities[job_id_priority] += 1  # Job Aging
                
    return schedule
