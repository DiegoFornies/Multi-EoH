
def heuristic(data):
    """
    A heuristic for solving the Flexible Job Shop Scheduling Problem (FJSP).
    Minimizes makespan, reduces time between operations in the same job,
    and balances machine load.

    Args:
        data (dict): A dictionary containing the problem instance data.

    Returns:
        dict: A dictionary representing the schedule.
    """

    n_jobs = data['n_jobs']
    n_machines = data['n_machines']
    jobs = data['jobs']

    # Initialize the schedule
    schedule = {}
    for job in jobs:
        schedule[job] = []

    # Keep track of machine availability
    machine_availability = {m: 0 for m in range(1, n_machines + 1)}

    # Keep track of the last end time for each job
    job_last_end_time = {j: 0 for j in range(1, n_jobs + 1)}

    # Iterate through each job and its operations
    for job_id, operations in jobs.items():
        for operation_index, operation_data in enumerate(operations):
            machines, processing_times = operation_data

            # Find the machine with the earliest availability and shortest processing time
            best_machine = None
            min_end_time = float('inf')

            for i, machine in enumerate(machines):
                processing_time = processing_times[i]
                start_time = max(machine_availability[machine], job_last_end_time[job_id])
                end_time = start_time + processing_time

                if end_time < min_end_time:
                    min_end_time = end_time
                    best_machine = machine
                    best_processing_time = processing_time
                    best_start_time = start_time

            # Update machine availability and job last end time
            machine_availability[best_machine] = min_end_time
            job_last_end_time[job_id] = min_end_time

            # Add the operation to the schedule
            schedule[job_id].append({
                'Operation': operation_index + 1,
                'Assigned Machine': str(best_machine),
                'Start Time': best_start_time,
                'End Time': min_end_time,
                'Processing Time': best_processing_time
            })

    return schedule
