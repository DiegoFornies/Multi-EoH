
def heuristic(input_data):
    """Hybrid heuristic: Earliest start, load balancing, and operation priority."""
    n_jobs = input_data['n_jobs']
    n_machines = input_data['n_machines']
    jobs = input_data['jobs']
    machine_available = {m: 0 for m in range(n_machines)}
    job_completion = {j: 0 for j in range(1, n_jobs + 1)}
    schedule = {}
    operations = []

    for job_id, job_ops in jobs.items():
        for op_index, op_data in enumerate(job_ops, 1):
            operations.append((job_id, op_index, op_data[0], op_data[1]))

    def calculate_priority(operation):
        """Prioritize operations by job completion time."""
        job_id, _, _, _ = operation
        return job_completion[job_id]

    operations.sort(key=calculate_priority)

    for job_id, op_index, possible_machines, possible_times in operations:
        best_machine, earliest_start, processing_time, best_makespan = None, float('inf'), None, float('inf')

        for m_index, machine_id in enumerate(possible_machines):
            process_time = possible_times[m_index]
            start_time = max(machine_available[machine_id], job_completion[job_id])
            makespan = start_time + process_time

            if makespan < best_makespan:
                best_makespan = makespan
                best_machine = machine_id
                earliest_start = start_time
                processing_time = process_time
            elif makespan == best_makespan and machine_available[machine_id] < (machine_available[best_machine] if best_machine is not None else float('inf')):
                best_machine = machine_id
                earliest_start = start_time
                processing_time = process_time
                best_makespan = makespan

        start_time = earliest_start
        end_time = start_time + processing_time
        if job_id not in schedule:
            schedule[job_id] = []

        schedule[job_id].append({
            'Operation': op_index,
            'Assigned Machine': best_machine,
            'Start Time': start_time,
            'End Time': end_time,
            'Processing Time': processing_time
        })

        machine_available[best_machine] = end_time
        job_completion[job_id] = end_time

    return schedule
