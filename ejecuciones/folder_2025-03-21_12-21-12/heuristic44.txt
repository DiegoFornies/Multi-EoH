
def heuristic(input_data):
    """Combines earliest start time with SPT and machine load for better scheduling."""
    n_jobs = input_data['n_jobs']
    n_machines = input_data['n_machines']
    jobs = input_data['jobs']

    machine_available_time = {m: 0 for m in range(n_machines)}
    job_completion_time = {j: 0 for j in range(1, n_jobs + 1)}
    machine_loads = {m: 0 for m in range(n_machines)}
    machine_assignments = {m: [] for m in range(n_machines)}

    solution = {}

    operation_queue = []
    for job, operations in jobs.items():
        operation_queue.append((job, 0))

    def calculate_priority(job, op_index):
        operation = jobs[job][op_index]
        machines, times = operation
        
        best_start = float('inf')
        best_machine = None
        best_time = None

        for i in range(len(machines)):
            machine = machines[i]
            processing_time = times[i]
            start_time = max(machine_available_time[machine], job_completion_time[job])
            
            if start_time < best_start:
                best_start = start_time
                best_machine = machine
                best_time = processing_time

        # Prioritize based on earliest start, shortest processing time, and machine load
        priority = best_start + (best_time / 1000) + (machine_loads[best_machine] / 100000)
        return priority, best_machine, best_time

    scheduled_operations = 0
    while scheduled_operations < sum(len(ops) for ops in jobs.values()):
        best_job = None
        best_op_index = None
        best_priority = float('inf')
        best_machine = None
        best_processing_time = None

        for job, op_index in operation_queue:
            priority, machine, time = calculate_priority(job, op_index)

            if priority < best_priority:
                best_priority = priority
                best_job = job
                best_op_index = op_index
                best_machine = machine
                best_processing_time = time
        
        if best_job is None:
            break

        operation = jobs[best_job][best_op_index]
        op_number = best_op_index + 1

        start_time = max(machine_available_time[best_machine], job_completion_time[best_job])
        end_time = start_time + best_processing_time

        if best_job not in solution:
            solution[best_job] = []

        solution[best_job].append({
            'Operation': op_number,
            'Assigned Machine': best_machine,
            'Start Time': start_time,
            'End Time': end_time,
            'Processing Time': best_processing_time
        })

        machine_assignments[best_machine].append((best_job,op_number,start_time,end_time))

        machine_available_time[best_machine] = end_time
        job_completion_time[best_job] = end_time
        machine_loads[best_machine] += best_processing_time

        operation_queue.remove((best_job, best_op_index))
        if best_op_index + 1 < len(jobs[best_job]):
            operation_queue.append((best_job, best_op_index + 1))
        
        scheduled_operations += 1
    return solution
